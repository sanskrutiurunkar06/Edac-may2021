Pyramid4

class Pyramid4
{
		public static void main(String args[])
		{
				for(int i = 1; i<=9; i++)
				{
					for(int j = 8; j>=i; j--)
					{
						System.out.print(" ");
					}
					for(int k = 1; k<=i; k++)
					{
						System.out.print(k);
					}
					for(int l = 2; l<=i; l++)
					{
						System.out.print(l-1);
					}
					
					System.out.println();
				}
		}
}



Pyramid5

class FullPyramid
{
	public static void main(String args[])
	{
		for(int i = 1; i<=6; i++)
		{
			for(int j = 10; j>=i; j-=2)
			{
				System.out.print(" ");
			}
			for(int j = 1; j<=i; j++)
				System.out.print("* ");
			System.out.println();
		}
	}
}
